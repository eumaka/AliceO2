46 template <typename T>
47 bool checkBranch(TTree& tree, BranchContent<T>&& content)
48 {
49  TBranch* branch = tree.GetBranch(content.branchName);
50  BOOST_REQUIRE(branch != nullptr);
51  T store;
52  T* pointer = &store;
53  // in general, pointer to pointer has to be used for setting the branch
54  // address to a store object, however this does not work for fundamental
55  // types, there the address to the variable has to be used in order
56  // to read back the value. Why? no clue.
57  if (std::is_fundamental<T>::value) {
58  branch->SetAddress(&store);
59  } else {
60  branch->SetAddress(&pointer);
61  }
62  branch->GetEntry(0);
63  BOOST_CHECK_MESSAGE(store == content.reference, "mismatch for branch " << content.branchName);
64  return store == content.reference;
65 }

104  // the branch names are simply built by adding the index

114  auto createPlainMessage = [&transport, &store](DataHeader&& dh, auto& data) {
115  dh.payloadSize = sizeof(data);
116  dh.payloadSerializationMethod = o2::header::gSerializationMethodNone;
117  DataProcessingHeader dph{ 0, 1 };
118  o2::header::Stack stack{ dh, dph };
119  FairMQMessagePtr header = transport->CreateMessage(stack.size());
120  FairMQMessagePtr payload = transport->CreateMessage(sizeof(data));
121  memcpy(header->GetData(), stack.data(), stack.size());
