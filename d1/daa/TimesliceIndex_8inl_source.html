30 inline bool TimesliceIndex::isValid(TimesliceSlot const& slot) const

67 inline TimesliceSlot TimesliceIndex::findOldestSlot() const
68 {
69  TimesliceSlot oldest{0};
70  auto oldPVal = std::get_if<uint64_t>(&mVariables[oldest.index].get(0));
71  if (oldPVal == nullptr) {
72  return oldest;
73  }
74  uint64_t oldTimestamp = *oldPVal;
75 
76  for (size_t i = 1; i < mVariables.size(); ++i) {
77  auto newPVal = std::get_if<uint64_t>(&mVariables[i].get(0));
78  if (newPVal == nullptr) {
79  return TimesliceSlot{i};
80  }
81  uint64_t newTimestamp = *newPVal;
82 
83  if (oldTimestamp > newTimestamp) {
84  oldest = TimesliceSlot{i};
85  oldTimestamp = newTimestamp;
86  }
87  }
88  return oldest;
89 }

107 inline TimesliceSlot TimesliceIndex::replaceLRUWith(data_matcher::VariableContext& newContext)
108 {
109  auto oldestSlot = findOldestSlot();
110  if (TimesliceIndex::isValid(oldestSlot) == false) {
111  mVariables[oldestSlot.index] = newContext;
112  return oldestSlot;
113  }
114  auto oldTimestamp = std::get_if<uint64_t>(&mVariables[oldestSlot.index].get(0));
115  if (oldTimestamp == nullptr) {
116  mVariables[oldestSlot.index] = newContext;
117  return oldestSlot;
118  }
119 
120  auto newTimestamp = std::get_if<uint64_t>(&newContext.get(0));
121  if (newTimestamp == nullptr) {
122  return {TimesliceSlot::INVALID};
123  }
124 
125  if (*newTimestamp > *oldTimestamp) {
126  mVariables[oldestSlot.index] = newContext;
127  return oldestSlot;
128  }
129  return {TimesliceSlot::INVALID};
130 }
