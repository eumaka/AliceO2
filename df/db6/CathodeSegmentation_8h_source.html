64  CathodeSegmentation(int detElemId, bool isBendingPlane)
65  : mImpl{ mchCathodeSegmentationConstruct(detElemId, isBendingPlane) },
66  mDualSampaIds{},
67  mDetElemId{ detElemId },
68  mIsBendingPlane{ isBendingPlane }
69  {
70  if (!mImpl) {
71  throw std::runtime_error("Can not create segmentation for DE " + std::to_string(detElemId) +
72  (mIsBendingPlane ? " Bending" : " NonBending"));
73  }
74  std::vector<int> dpid;
75  auto addDualSampaId = [&dpid](int dualSampaId) { dpid.push_back(dualSampaId); };
76  auto callback = [](void* data, int dualSampaId) {
77  auto fn = static_cast<decltype(&addDualSampaId)>(data);
78  (*fn)(dualSampaId);
79  };
80  mchCathodeSegmentationForEachDualSampa(mImpl, callback, &addDualSampaId);
81  mDualSampaIds = dpid;
82  int n{ 0 };
83  for (auto i = 0; i < nofDualSampas(); ++i) {
84  forEachPadInDualSampa(dualSampaId(i), [&n](int /*catPadIndex*/) { ++n; });
85  }
86  mNofPads = n;
87  }
