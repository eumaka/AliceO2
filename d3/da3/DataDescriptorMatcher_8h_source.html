66 class VariableContext
67 {
68  public:
69  VariableContext()
70  : mPerformedUpdates{ 0 }
71  {
72  }
73 
74  ContextElement::Value const& get(size_t pos) const
75  {
76  // First we check if there is any pending update
77  for (size_t i = 0; i < mPerformedUpdates; ++i) {
78  if (mUpdates[i].position == pos) {
79  return mUpdates[i].newValue;
80  }
81  }
82  // Otherwise we return the element.
83  return mElements.at(pos).value;
84  }
85 
86  void put(ContextUpdate&& update)
87  {
88  mUpdates[mPerformedUpdates++] = std::move(update);
89  }
90 
95  void commit()
96  {
97  for (size_t i = 0; i < mPerformedUpdates; ++i) {
98  mElements[mUpdates[i].position].value = mUpdates[i].newValue;
99  }
100  mPerformedUpdates = 0;
101  }
102 
105  void discard()
106  {
107  mPerformedUpdates = 0;
108  }
109 
112  void reset()
113  {
114  mPerformedUpdates = 0;
115  for (auto& element : mElements) {
116  element.value = None{};
117  }
118  }
119 
120  private:
121  /* We make this class fixed size to avoid memory churning while
122  matching as much as posible when doing the matching, as that might become
123  performance critical. Given we will have only a few of these (one per
124  cacheline of the input messages) it should not be critical memory wise.
125  */
126  std::array<ContextElement, MAX_MATCHING_VARIABLE> mElements;
127  std::array<ContextUpdate, MAX_UPDATES_PER_QUERY> mUpdates;
128  int mPerformedUpdates;
129 };

194  bool match(header::DataHeader const& header, VariableContext& context) const
195  {
196  if (auto ref = std::get_if<ContextRef>(&mValue)) {
197  auto& variable = context.get(ref->index);
198  if (auto value = std::get_if<std::string>(&variable)) {
199  return strncmp(header.dataOrigin.str, value->c_str(), 4) == 0;
200  }
201  auto maxSize = strnlen(header.dataOrigin.str, 4);
202  context.put({ ref->index, std::string(header.dataOrigin.str, maxSize) });
203  return true;
204  } else if (auto s = std::get_if<std::string>(&mValue)) {
205  return strncmp(header.dataOrigin.str, s->c_str(), 4) == 0;
206  }
207  throw std::runtime_error("Mismatching type for variable");
208  }
