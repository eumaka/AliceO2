45 static float binner(BinFunction binFunction,
46  const void* const* datas,
47  size_t data_idx,
48  size_t values_count,
49  float binStart, float binEnd,
50  size_t domain_min, size_t domain_max,
51  float (*getterY)(const void* data, int idx),
52  size_t (*getterX)(const void* data, int idx),
53  size_t& cacheXIdx,
54  size_t& count)
55 {
56  float value;
57  switch (binFunction) {
58  case BinFunction::CEIL:
59  case BinFunction::MAX:
60  value = -FLT_MAX;
61  break;
62  case BinFunction::FLOOR:
63  case BinFunction::MIN:
64  value = FLT_MAX;
65  break;
66  default:
67  value = 0;
68  }
69  float scale = 1.f / (domain_max - domain_min);
70  // Accumulate the matching datapoints.
71  while (cacheXIdx < values_count) {
72  size_t bin = getterX(datas[data_idx], cacheXIdx);
73  if ((bin < domain_min)) {
74  cacheXIdx++;
75  continue;
76  }
77  if ((bin > domain_max)) {
78  return value;
79  }
80  // We are withing (domain_min, domain_max), what we have have below
81  // works.
82  float binPos = ImSaturate(float(bin - domain_min) * scale);
83  if ((binPos < binStart)) {
84  cacheXIdx++;
85  continue;
86  }
87  if ((binPos > binEnd)) {
88  return value;
89  }
90  count++;
91  float binValue = getterY(datas[data_idx], cacheXIdx);
92  cacheXIdx++;
93  switch (binFunction) {
94  case BinFunction::CEIL:
95  value = std::max(value, ceilf(binValue));
96  break;
97  case BinFunction::MAX:
98  value = std::max(value, binValue);
99  break;
100  case BinFunction::FLOOR:
101  value = std::min(value, floorf(binValue));
102  break;
103  case BinFunction::MIN:
104  value = std::min(value, binValue);
105  break;
106  case BinFunction::COUNT:
107  value = count;
108  break;
109  case BinFunction::AVG:
110  // NOTE: this will not actually work with too many values because of floating point
111  value = (value * (count - 1) / count) + (binValue / count);
112  break;
113  }
114  }
115  return value;
116 }
