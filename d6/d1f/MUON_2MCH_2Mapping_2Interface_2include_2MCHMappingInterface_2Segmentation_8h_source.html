169 inline int Segmentation::findPadByFEE(int dualSampaId, int dualSampaChannel) const
170 {
171  bool isBending = dualSampaId < 1024;
172  int catPadIndex;
173  if (isBending) {
174  catPadIndex = mBending.findPadByFEE(dualSampaId, dualSampaChannel);
175  if (!mBending.isValid(catPadIndex)) {
176  return -1;
177  }
178  } else {
179  catPadIndex = mNonBending.findPadByFEE(dualSampaId, dualSampaChannel);
180  if (!mNonBending.isValid(catPadIndex)) {
181  return -1;
182  }
183  }
184  return padC2DE(catPadIndex, isBending);
185 }

218 inline bool Segmentation::findPadPairByPosition(double x, double y, int& b, int& nb) const
219 {
220  b = mBending.findPadByPosition(x, y);
221  nb = mNonBending.findPadByPosition(x, y);
222  if (!mBending.isValid(b) || !mNonBending.isValid(nb)) {
223  return false;
224  }
225  b = padC2DE(b, true);
226  nb = padC2DE(nb, false);
227  return true;
228 }
