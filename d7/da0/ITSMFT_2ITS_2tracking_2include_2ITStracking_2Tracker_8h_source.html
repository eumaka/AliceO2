132  T&&... args)
133 {
134  float diff{ 0.f };
135 
136  if (Constants::DoTimeBenchmarks) {
137  auto start = std::chrono::high_resolution_clock::now();
138  (this->*task)(std::forward<T>(args)...);
139  auto end = std::chrono::high_resolution_clock::now();
140 
141  std::chrono::duration<double, std::milli> diff_t{ end - start };
142  diff = diff_t.count();
143 
144  if (taskName == nullptr) {
145  ostream << diff << "\t";
146  } else {
147  ostream << std::setw(2) << " - " << taskName << " completed in: " << diff << " ms" << std::endl;
148  }
149  } else {
150  (this->*task)(std::forward<T>(args)...);
151  }
152 
153  return diff;
154 }
