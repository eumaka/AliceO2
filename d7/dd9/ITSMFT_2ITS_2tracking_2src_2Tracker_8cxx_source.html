115  iNextLayerCell < nextLayerCellsNum &&

117  nextLayerTrackletIndex;

121  const float3 currentCellNormalVector{ currentCell.getNormalVectorCoordinates() };

379 bool Tracker::fitTrack(const ROframe& event, TrackITS& track, int start, int end, int step)
380 {
381  track.setChi2(0);
382  for (int iLayer{ start }; iLayer != end; iLayer += step) {
383  if (track.getClusterIndex(iLayer) == Constants::ITS::UnusedIndex) {
384  continue;
385  }
386  const TrackingFrameInfo& trackingHit = event.getTrackingFrameInfoOnLayer(iLayer).at(track.getClusterIndex(iLayer));
387 
388  if (!track.rotate(trackingHit.alphaTrackingFrame))
389  return false;
390 
391  if (!track.propagateTo(trackingHit.xTrackingFrame, getBz()))
392  return false;
393 
394  track.setChi2(track.getChi2() +
395  track.getPredictedChi2(trackingHit.positionTrackingFrame, trackingHit.covarianceTrackingFrame));
396  if (!track.TrackParCov::update(trackingHit.positionTrackingFrame, trackingHit.covarianceTrackingFrame))
397  return false;
398 
399  const float xx0 = (iLayer > 2) ? 0.008f : 0.003f; // Rough layer thickness
400  constexpr float radiationLength = 9.36f; // Radiation length of Si [cm]
401  constexpr float density = 2.33f; // Density of Si [g/cm^3]
402  if (!track.correctForMaterial(xx0, xx0 * radiationLength * density, true))
403  return false;
404  }
405  return true;
406 }

565 track::TrackParCov Tracker::buildTrackSeed(const Cluster& cluster1, const Cluster& cluster2,
566  const Cluster& cluster3, const TrackingFrameInfo& tf3)
567 {
568  const float ca = std::cos(tf3.alphaTrackingFrame), sa = std::sin(tf3.alphaTrackingFrame);
569  const float x1 = cluster1.xCoordinate * ca + cluster1.yCoordinate * sa;
570  const float y1 = -cluster1.xCoordinate * sa + cluster1.yCoordinate * ca;
571  const float z1 = cluster1.zCoordinate;
572  const float x2 = cluster2.xCoordinate * ca + cluster2.yCoordinate * sa;
573  const float y2 = -cluster2.xCoordinate * sa + cluster2.yCoordinate * ca;
574  const float z2 = cluster2.zCoordinate;
575  const float x3 = tf3.xTrackingFrame;
576  const float y3 = tf3.positionTrackingFrame[0];
577  const float z3 = tf3.positionTrackingFrame[1];
578 
579  const float crv = MathUtils::computeCurvature(x1, y1, x2, y2, x3, y3);
580  const float x0 = MathUtils::computeCurvatureCentreX(x1, y1, x2, y2, x3, y3);
581  const float tgl12 = MathUtils::computeTanDipAngle(x1, y1, x2, y2, z1, z2);
582  const float tgl23 = MathUtils::computeTanDipAngle(x2, y2, x3, y3, z2, z3);
583 
584  const float fy = 1. / (cluster2.rCoordinate - cluster3.rCoordinate);
585  const float& tz = fy;
