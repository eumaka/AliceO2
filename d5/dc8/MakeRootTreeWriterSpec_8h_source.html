263  auto initFct = [processAttributes, TerminationPolicyMap = TerminationPolicyMap](InitContext& ic) {
264  auto& branchNameOptions = processAttributes->branchNameOptions;

292  auto processingFct = [processAttributes](ProcessingContext& pc) {
293  auto& writer = processAttributes->writer;
294  auto& terminationPolicy = processAttributes->terminationPolicy;
295  auto& terminationCondition = processAttributes->terminationCondition;
296  auto& activeInputs = processAttributes->activeInputs;
297  auto& counter = processAttributes->counter;
298  auto& nEvents = processAttributes->nEvents;
299  if (writer->isClosed()) {
300  return;
301  }
302 
303  // if the termination condition contains function of type CheckProcessing, this is checked
304  // before the processing, currently implemented logic:
305  // - processing is skipped if this is indicated at least one input
306  // - treated as 'ready' if this is indicated at least one input
307  auto checkProcessing = [&terminationCondition, &activeInputs](auto const& inputs) {
308  bool doProcessing = true;
309  if (std::holds_alternative<TerminationCondition::CheckProcessing>(terminationCondition.check)) {
310  auto& check = std::get<TerminationCondition::CheckProcessing>(terminationCondition.check);
311  for (auto const& ref : inputs) {
312  bool ready = false;
313  auto iter = activeInputs.find(ref.spec->binding);
314  if (check(ref, ready) == TerminationCondition::Action::SkipProcessing) {
315  // this condition tells us not to process the data any further
316  doProcessing = false;
317  }
318  if (iter != activeInputs.end() && ready) {
319  // this input is ready, remove from active inputs
320  activeInputs.erase(iter);
321  }
322  }
323  }
324  return doProcessing;
325  };
326  // if the termination condition contains function of type CheckReady, this is checked
327  // after the processing, treated as 'ready' if indicated by at least one input
328  auto checkReady = [&terminationCondition, &activeInputs](auto const& inputs) {
329  if (std::holds_alternative<TerminationCondition::CheckReady>(terminationCondition.check)) {
330  auto& check = std::get<TerminationCondition::CheckReady>(terminationCondition.check);
331  for (auto const& ref : inputs) {
332  auto iter = activeInputs.find(ref.spec->binding);
333  if (iter != activeInputs.end() && check(ref)) {
334  // this input is ready, remove from active inputs
335  activeInputs.erase(iter);
336  }
337  }
338  }
339  return activeInputs.size() == 0;
340  };
341 
342  if (checkProcessing(pc.inputs())) {
343  (*writer)(pc.inputs());
344  counter = counter + 1;
345  }
346 
347  if ((nEvents >= 0 && counter == nEvents) || checkReady(pc.inputs())) {
348  writer->close();
349  pc.services().get<ControlService>().readyToQuit(terminationPolicy == TerminationPolicy::Workflow);
350  }
351  };
352 
353  return processingFct;
354  };
