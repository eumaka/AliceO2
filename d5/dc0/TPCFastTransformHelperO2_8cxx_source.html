77

94  distortion.setTPCgeometry(tpcZlengthSideA, tpcZlengthSideC);
95 
96  for (int iRow = 0; iRow < fastTransform.getNumberOfRows(); iRow++) {
97  Sector sector = 0;
98  int regionNumber = 0;
99  while (iRow >= mapper.getGlobalRowOffsetRegion(regionNumber) + mapper.getNumberOfRowsRegion(regionNumber))
100  regionNumber++;
101

108  const PadCentre& padCentre = mapper.padCentre(pad);
109  float xRow = padCentre.X();

111  fastTransform.setTPCrow(iRow, xRow, nPads, padWidth);
112  distortion.setTPCrow(iRow, xRow, nPads, padWidth, 0);
113  }
114 
115  // set some initial values, will be reinitialised later int updateCalibration()
116  const double t0 = 0.;
117  const double vdCorrY = 0.;
118  const double ldCorr = 0.;
119  const double tpcAlignmentZ = 0.;
120  const double tofCorr = 0.;
121  const double primVtxZ = 0.;
122  const double initTimeStamp = -1;
123  fastTransform.setCalibration(initTimeStamp, t0, vDrift, vdCorrY, ldCorr, tofCorr, primVtxZ, tpcAlignmentZ);
124 
125  IrregularSpline2D3D spline;
126  {
127  int nKnotsU = 15;
128  int nAxisTicksU = mapper.getNumberOfPadsInRowSector(10);
129  int nKnotsV = 20;
130  int nAxisTicksV = lastTimeBin + 1;
131  float knotsU[nKnotsU];
132  float knotsV[nKnotsV];
133  for (int i = 0; i < nKnotsU; i++)
134  knotsU[i] = 1. / (nKnotsU - 1) * i;
135  for (int i = 0; i < nKnotsV; i++)
136  knotsV[i] = 1. / (nKnotsV - 1) * i;
137 
138  // TODO: adjust the grid
139 
140  double d1 = 0.6;
141  double d2 = 0.9 - d1;
142  double d3 = 1. - d2 - d1;
143 
144  for (int i = 0; i < 5; i++) { // 5 bins in first 6% of drift
145  knotsV[i] = i / 4. * d1;
146  }
147  for (int i = 0; i < 10; i++) { // 10 bins for 6% <-> 90%
148  knotsV[4 + i] = d1 + i / 9. * d2;
149  }
150  for (int i = 0; i < 5; i++) { // 5 bins for last 90% <-> 100%
151  knotsV[13 + i] = d1 + d2 + i / 4. * d3;
152  }
153 
154  spline.construct(nKnotsU, knotsU, nAxisTicksU,
155  nKnotsV, knotsV, nAxisTicksV);
156  }
157  distortion.setApproximationScenario(0, spline);
158  distortion.finishConstruction();
159  fastTransform.finishConstruction();
160 
161  // check if calculated pad geometry is consistent with the map
162  testGeometry(fastTransform);
163 
164  updateCalibration(fastTransform, TimeStamp);
165 
166  return std::move(fastTransformPtr);

169 int TPCFastTransformHelperO2::updateCalibration(ali_tpc_common::tpc_fast_transformation::TPCFastTransform& fastTransform, Long_t TimeStamp)
170 {
171  // Update the calibration with the new time stamp
172 
173  if (!mIsInitialized) {
174  init();
175  }
176 
177  Long_t lastTS = fastTransform.getTimeStamp();
178 
179  // deinitialize
180 
181  fastTransform.setTimeStamp(-1);
182 
183  if (TimeStamp < 0) {
184  return 0;
185  }
186 
187  // search for the calibration database ...
188 
189  const static ParameterDetector& detParam = ParameterDetector::defaultInstance();
190  const static ParameterGas& gasParam = ParameterGas::defaultInstance();
191  const static ParameterElectronics& elParam = ParameterElectronics::defaultInstance();
192 
193  // calibration found, set the initialized status back
194 
195  fastTransform.setTimeStamp(lastTS);
196 
197  // less than 60 seconds from the previois time stamp, don't do anything
198 
199  if (lastTS >= 0 && TMath::Abs(lastTS - TimeStamp) < 60) {
200  return 0;
201  }
202 
203  // start the initialization
204 
205  fastTransform.setTimeStamp(TimeStamp);
206 
207  // find last calibrated time bin
208 
209  const double vDrift = elParam.getZBinWidth() * gasParam.getVdrift();
210 
211  //mLastTimeBin = detParam.getTPClength() / vDrift + 1;
212 
213  // fast transform formula:
214  // L = (t-t0)*(mVdrift + mVdriftCorrY*yLab ) + mLdriftCorr
215  // Z = Z(L) + tpcAlignmentZ
216  // spline distortions for xyz
217  // Time-of-flight correction: ldrift += dist-to-vtx*tofCorr
218 
219  const double t0 = 0.;
220  const double vdCorrY = 0.;
221  const double ldCorr = 0.;
222  const double tpcAlignmentZ = 0.;
223 
224  const double tofCorr = 0.;
225  const double primVtxZ = 0.;
226 
227  fastTransform.setCalibration(TimeStamp, t0, vDrift, vdCorrY, ldCorr, tofCorr, primVtxZ, tpcAlignmentZ);
228 
229  // now calculate distortion map: dx,du,dv = ( origTransform() -> x,u,v) - fastTransformNominal:x,u,v
230 
231  TPCDistortionIRS& distortion = fastTransform.getDistortionNonConst();
232 
233  // switch TOF correction off for a while
234 
235  for (int slice = 0; slice < distortion.getNumberOfSlices(); slice++) {
236  for (int row = 0; row < distortion.getNumberOfRows(); row++) {
237  //const TPCFastTransform::RowInfo &rowInfo = fastTransform.getRowInfo( row );
238  const IrregularSpline2D3D& spline = distortion.getSpline(slice, row);
239  float* data = distortion.getSplineDataNonConst(slice, row);
240  for (int knot = 0; knot < spline.getNumberOfKnots(); knot++) {
241  data[3 * knot + 0] = 0.f;
242  data[3 * knot + 1] = 0.f;
243  data[3 * knot + 2] = 0.f;
244  } // knots
245  spline.correctEdges(data);
246  } // row
247  } // slice
248 
249  // set back the time-of-flight correction
250 
251  return 0;
252 }

254 void TPCFastTransformHelperO2::testGeometry(const ali_tpc_common::tpc_fast_transformation::TPCFastTransform& fastTransform) const
255 {
256  const Mapper& mapper = Mapper::instance();
257 
258  if (fastTransform.getNumberOfSlices() != Sector::MAXSECTOR) {
259  LOG(FATAL) << "Wrong number of sectors :" << fastTransform.getNumberOfSlices() << " instead of " << Sector::MAXSECTOR << std::endl;
260  }
261 
262  if (fastTransform.getNumberOfRows() != mapper.getNumberOfRows()) {
263  LOG(FATAL) << "Wrong number of rows :" << fastTransform.getNumberOfRows() << " instead of " << mapper.getNumberOfRows() << std::endl;
264  }
265 
266  double maxDx = 0, maxDy = 0;
267 
268  for (int row = 0; row < fastTransform.getNumberOfRows(); row++) {
269 
270  const int nPads = fastTransform.getRowInfo(row).maxPad + 1;
271 
272  if (nPads != mapper.getNumberOfPadsInRowSector(row)) {
273  LOG(FATAL) << "Wrong number of pads :" << nPads << " instead of " << mapper.getNumberOfPadsInRowSector(row) << std::endl;
274  }
275 
276  const double x = fastTransform.getRowInfo(row).x;
277 
278  // check if calculated pad positions are equal to the real ones
279 
280  for (int pad = 0; pad < nPads; pad++) {
281  const GlobalPadNumber p = mapper.globalPadNumber(PadPos(row, pad));
282  const PadCentre& c = mapper.padCentre(p);
283  float u = 0, v = 0;
284  int err = fastTransform.convPadTimeToUV(0, row, pad, 10., u, v, 0.);
285  if (err != 0) {
286  LOG(FATAL) << "Can not transform a cluster: row " << row << " pad " << pad << " time 10. : error " << err << std::endl;
287  }
288 
289  const double dx = x - c.X();
290  const double dy = u - (-c.Y()); // diferent sign convention for Y coordinate in the map
291 
292  if (fabs(dx) >= 1.e-6 || fabs(dy) >= 1.e-5) {
293  LOG(WARNING) << "wrong calculated pad position:"
294  << " row " << row << " pad " << pad << " x calc " << x << " x in map " << c.X() << " dx " << (x - c.X())
295  << " y calc " << u << " y in map " << -c.Y() << " dy " << dy << std::endl;
296  }
297  if (fabs(maxDx) < fabs(dx)) {
298  maxDx = dx;
299  }
300  if (fabs(maxDy) < fabs(dy)) {
301  maxDy = dy;
302  }
303  }
304  }
305 
306  if (fabs(maxDx) >= 1.e-4 || fabs(maxDy) >= 1.e-4) {
307  LOG(FATAL) << "wrong calculated pad position:"
308  << " max Dx " << maxDx << " max Dy " << maxDy << std::endl;
309  }
310 }
