151 inline float SAMPAProcessing::makeSignal(float ADCcounts, const CRU& cru, const PadPos& pos,
152  float& pedestal, float& noise)
153 {
154  float signal = ADCcounts;
155  pedestal = getPedestal(cru, pos);
156  noise = getNoise(cru, pos);
157  switch (mEleParam->getDigitizationMode()) {
158  case DigitzationMode::FullMode: {
159  signal += noise;
160  signal += pedestal;
161  return getADCSaturation(signal);
162  break;
163  }
164  case DigitzationMode::SubtractPedestal: {
165  signal += noise;
166  signal += pedestal;
167  float signalSubtractPedestal = getADCSaturation(signal) - pedestal;
168  return signalSubtractPedestal;
169  break;
170  }
171  case DigitzationMode::NoSaturation: {
172  signal += noise;
173  signal += pedestal;
174  return signal;
175  break;
176  }
177  case DigitzationMode::PropagateADC: {
178  return signal;
179  break;
180  }
181  }
182 }

184 inline float SAMPAProcessing::getADCSaturation(const float signal) const
185 {
187  const float saturatedSignal = mSaturationSpline->Eval(signal);
188  const float adcSaturation = mEleParam->getADCSaturation();
189  if (saturatedSignal > adcSaturation - 1)
190  return adcSaturation - 1;
191  return saturatedSignal;
192 }

194 template <typename T>
195 inline T SAMPAProcessing::getGamma4(T time, T startTime, T ADC) const
196 {
197  Vc::float_v tmp0 = (time - startTime) / mEleParam->getPeakingTime();
198  Vc::float_m cond = (tmp0 > 0);
199  Vc::float_v tmp;
200  tmp(cond) = tmp0;
201  Vc::float_v tmp2 = tmp * tmp;
202  return 55.f * ADC * Vc::exp(-4.f * tmp) * tmp2 * tmp2;
203 }
